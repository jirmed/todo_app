cmake_minimum_required(VERSION 3.25)

project(todo_app)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Optimalizace podle kompileru a build typu
if(CMAKE_BUILD_TYPE STREQUAL "Debug")
    if(MSVC)
        set(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} /Od")
        set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /MP")  # Paraleln칤 build pro MSVC
        set(CMAKE_EXE_LINKER_FLAGS_DEBUG "${CMAKE_EXE_LINKER_FLAGS_DEBUG} /INCREMENTAL")
    else()
        set(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} -O1")
    endif()
    # Ponech치me statick칠 linkov치n칤 pro konzistenci
    set(BUILD_SHARED_LIBS OFF)
endif()

include(FetchContent)

# Cachov치n칤 z치vislost칤
set(FETCHCONTENT_QUIET OFF)
set(FETCHCONTENT_BASE_DIR "${CMAKE_BINARY_DIR}/_deps" CACHE PATH "FetchContent base directory")

# 游닍 nlohmann::json
FetchContent_Declare(
    json
    URL https://github.com/nlohmann/json/releases/download/v3.11.2/json.tar.xz
    DOWNLOAD_EXTRACT_TIMESTAMP TRUE
)
FetchContent_MakeAvailable(json)

# 游닍 Asio (standalone)
FetchContent_Declare(
        asio
        GIT_REPOSITORY https://github.com/chriskohlhoff/asio.git
        GIT_TAG asio-1-28-1
        DOWNLOAD_EXTRACT_TIMESTAMP TRUE
)
FetchContent_MakeAvailable(asio)

# Nastaven칤 ASIO_INCLUDE_DIR pro FindAsio.cmake modul v Crow
set(ASIO_INCLUDE_DIR ${asio_SOURCE_DIR}/asio/include)
set(ASIO_FOUND TRUE)

# Vytvo콏en칤 asio::asio targetu pro p콏칤pad, 쬰 ho Crow nenalezne
if(NOT TARGET asio::asio)
    add_library(asio::asio INTERFACE IMPORTED)
    target_include_directories(asio::asio INTERFACE ${ASIO_INCLUDE_DIR})
    target_compile_definitions(asio::asio INTERFACE ASIO_STANDALONE)
    find_package(Threads REQUIRED)
    target_link_libraries(asio::asio INTERFACE Threads::Threads)
endif()

# 游닍 Crow framework
FetchContent_Declare(
        crow
        GIT_REPOSITORY https://github.com/CrowCpp/crow.git
        GIT_TAG v1.2.1.2
        DOWNLOAD_EXTRACT_TIMESTAMP TRUE
)

# Nastaven칤 prom캩nn칳ch pro Crow p콏ed jeho na캜ten칤m
set(CROW_USE_BOOST OFF CACHE BOOL "Use Boost instead of standalone asio" FORCE)
set(CROW_BUILD_TESTS OFF CACHE BOOL "Build Crow tests" FORCE)
set(CROW_BUILD_EXAMPLES OFF CACHE BOOL "Build Crow examples" FORCE)

FetchContent_MakeAvailable(crow)

# 游닍 GoogleTest - explicitn캩 nastav칤me statick칠 linkov치n칤
FetchContent_Declare(
        googletest
        URL https://github.com/google/googletest/archive/refs/heads/main.zip
        DOWNLOAD_EXTRACT_TIMESTAMP TRUE
)
set(INSTALL_GTEST OFF CACHE BOOL "Install GoogleTest" FORCE)
set(BUILD_GMOCK OFF CACHE BOOL "Build GoogleMock" FORCE)
set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
FetchContent_MakeAvailable(googletest)

# 游닄 Core knihovna - zm캩n칤me na statickou
# Rekurzivn캩 najdeme v코echny zdrojov칠 soubory v adres치콏i src
file(GLOB_RECURSE SRC_FILES 
    "${CMAKE_CURRENT_SOURCE_DIR}/src/**/*.cpp"
)

# P콏id치me explicitn캩 chyb캩j칤c칤 soubor
list(APPEND SRC_FILES "${CMAKE_CURRENT_SOURCE_DIR}/src/presentation/http/RestServer.cpp")

# Vylou캜칤me main.cpp ze zdrojov칳ch soubor콢 knihovny
list(FILTER SRC_FILES EXCLUDE REGEX ".*main\\.cpp$")

add_library(core STATIC ${SRC_FILES})

# Precompiled headers pro rychlej코칤 kompilaci
target_precompile_headers(core
    PUBLIC
    <iostream>
    <string>
    <vector>
    <memory>
    <unordered_map>
    <map>
    <algorithm>
    <nlohmann/json.hpp>
)

target_include_directories(core
    PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/src
    ${ASIO_INCLUDE_DIR}
)

target_link_libraries(core
    PUBLIC
    nlohmann_json::nlohmann_json
    Crow::Crow
)

target_compile_definitions(core
    PUBLIC
    ASIO_STANDALONE
)

# Linkov치n칤 Windows specifick칳ch knihoven
if(WIN32)
    target_link_libraries(core
        PUBLIC
        ws2_32
        mswsock
    )
endif()

# 游댢 Hlavn칤 aplikace
add_executable(app src/main.cpp)
target_link_libraries(app PRIVATE core)

# 游빍 Testy
include(GoogleTest)
enable_testing()

# Najd캩te v코echny testovac칤 soubory (ponechte jak je)
file(GLOB_RECURSE TEST_FILES
    "${CMAKE_CURRENT_SOURCE_DIR}/tests/**/*_test.cpp"
)

# Vytvo콏te samostatn칠 test executables pro ka쬯칳 testovac칤 soubor
foreach(TEST_FILE ${TEST_FILES})
    # Z칤skat n치zev testu bez cesty a p콏칤pony
    get_filename_component(TEST_NAME ${TEST_FILE} NAME_WE)

    # P콏idat samostatn칳 test executable
    add_executable(${TEST_NAME} ${TEST_FILE})

    target_include_directories(${TEST_NAME}
        PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/tests
    )

    target_link_libraries(${TEST_NAME}
        PRIVATE
        core
        gtest_main
        gtest
    )

#    # Windows output adres치콏 pro testy
    if(WIN32 AND CMAKE_BUILD_TYPE STREQUAL "Debug")
        set_target_properties(${TEST_NAME} PROPERTIES
            RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
        )
    endif()

    # Pou쬴t칤 gtest_discover_tests m칤sto add_test pro detekci jednotliv칳ch testovac칤ch p콏칤pad콢
    gtest_discover_tests(${TEST_NAME}
        WORKING_DIRECTORY "$<TARGET_FILE_DIR:${TEST_NAME}>"
        PROPERTIES
            VS_DEBUGGER_WORKING_DIRECTORY "$<TARGET_FILE_DIR:${TEST_NAME}>"
        TEST_PREFIX "${TEST_NAME}."
        TEST_LIST ${TEST_NAME}_TESTS
    )

    # Nastaven칤 vlastnosti TEST_DISCOVERY_TIMEOUT pro del코칤 dobu objevov치n칤 test콢
    set_property(GLOBAL PROPERTY TEST_DISCOVERY_TIMEOUT 60)
endforeach()

# Ponech치me i p콢vodn칤 unit_tests executable pro zp캩tnou kompatibilitu
add_executable(unit_tests ${TEST_FILES})

target_include_directories(unit_tests
    PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/tests/unit
)

target_link_libraries(unit_tests
    PRIVATE
    core
    gtest_main
    gtest
)

# Windows output adres치콏 pro testy
if(WIN32 AND CMAKE_BUILD_TYPE STREQUAL "Debug")
    set_target_properties(unit_tests PROPERTIES
        RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
    )
endif()
